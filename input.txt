fn fib(x: int) -> int {
    if x<=1 {
        return 1;
    }
    let result: int = fib(x - 1);
    result = result + fib(x - 2);
    return result;
}

fn main() -> int {
    let i: int = 0;
    let j: int;
    i = i+(i=1);
    while i < j {
        fib(i);
        let i : int = 1;
        i = i + 1;
    }
    return 0;
}
